# Git Basic Commands

1. git init -- To initialize an empty Git repository
2. git status -- To check current repository status

# Add Commands
3. git add . -- To stage all files
4. git add <file> -- To stage specific file
5. git add *.ext -- To stage matching pattern

# Commit Commands
6. git commit -m "msg" -- Commit staged changes
7. git commit -am "msg" -- Add + commit tracked files

# Log Commands
8. git log -- To see detailed commit history
9. git log --oneline -- Short commit history
10. git log --graph --decorate --oneline --all -- Visual commit tree
