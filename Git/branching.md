# Git Branching Commands

1. git branch -- To list all branches
2. git branch <name> -- To create branch
3. git branch -d <name> -- Delete merged branch safely
4. git branch -D <name> -- Force delete branch

# Checkout
5. git checkout <branch> -- Switch branch
6. git checkout -b <branch> -- Create + switch

# Modern Alternative (Recommended)
7. git switch <branch> -- Switch branch
8. git switch -c <branch> -- Create + switch
