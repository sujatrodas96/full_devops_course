# Push / Pull / Fetch Commands

# Push
1. git push -- Push commits to remote
2. git push origin main -- Push to main branch
3. git push -u origin main -- Set upstream

# Pull
4. git pull -- Fetch + merge
5. git pull origin main -- Pull main branch
6. git pull --rebase -- Pull using rebase

# Fetch
7. git fetch -- Download changes without merging
8. git fetch origin -- Fetch from origin
