# SYSTEM INFO & PROCESS MANAGEMENT

uname -a -- System info
hostname -- Hostname
hostnamectl -- OS + kernel
date -- Date
cal -- Calendar

df -h -- Disk usage
du -sh -- Folder size
du -sh * -- Sizes

free -h -- Memory
top -- Running processes
htop -- Better process viewer
ps aux -- All processes
kill PID -- Kill process
kill -9 PID -- Force kill
