# BASIC LINUX COMMANDS

1. pwd -- Show current working directory
2. ls -- List files and directories
3. ls -l -- Detailed list
4. ls -a -- Show hidden files
5. ls -lh -- Human readable size

6. cd <dir> -- Change directory
7. cd .. -- Go back
8. cd / -- Root directory
9. cd ~ -- Home directory

10. mkdir <name> -- Create directory
11. mkdir -p a/b/c -- Nested directories

12. touch <file> -- Create file
13. rm <file> -- Remove file
14. rm -r <dir> -- Remove directory
15. rm -rf <dir> -- Force delete directory

16. cp source dest -- Copy file
17. cp -r source dest -- Copy folder
18. mv source dest -- Move / rename

19. cat file -- Show content
20. tac file -- Reverse content
21. nl file -- Show line numbers
22. head file -- First 10 lines
23. tail file -- Last 10 lines
24. tail -f file -- Live logs

25. file filename -- Check file type
26. which cmd -- Command location
27. whereis cmd -- Binary + doc path
